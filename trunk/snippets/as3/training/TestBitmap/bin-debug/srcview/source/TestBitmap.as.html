package
{
    import flash.display.Bitmap;
    import flash.display.BitmapData;
    import flash.display.Sprite;
    
    public class TestBitmap extends Sprite
    {
        public function TestBitmap()
        {
            var data:BitmapData =  new BitmapData(100, 100, true, 0xffff0000);
            var bitmap:Bitmap = new Bitmap(data);
            var bitmap2:Bitmap = new Bitmap(data.clone());
            bitmap2.y = 120;
            bitmap2.rotation = 45;
            addChild(bitmap);
            addChild(bitmap2);
            for(var i:int = 40; i<60; i++){
                for(var j:int = 40; j<60; j++){
                    data.setPixel(i, j, 0x00FF00);
                }
            }
        }
    }
}
